########################################################################
# Check if there is C++ code at all
########################################################################
if(NOT scapy_radio_sources)
    MESSAGE(STATUS "No C++ sources... skipping python bindings")
    return()
endif(NOT scapy_radio_sources)

########################################################################
# Check for pygccxml
########################################################################
GR_PYTHON_CHECK_MODULE_RAW(
    "pygccxml"
    "import pygccxml"
    PYGCCXML_FOUND
    )

include(GrPybind)

########################################################################
# Python Bindings
########################################################################

list(APPEND scapy_radio_python_files
        add_gr_header_python.cc
        strip_gr_header_python.cc
    python_bindings.cc)

GR_PYBIND_MAKE_OOT(scapy_radio
   ../../..
   gr::scapy_radio
   "${scapy_radio_python_files}")

# copy in bindings .so file for use in QA test module
add_custom_target(
  copy_bindings_for_tests ALL
  COMMAND
    ${CMAKE_COMMAND} -E copy "${CMAKE_CURRENT_BINARY_DIR}/*.so"
    ${CMAKE_BINARY_DIR}/test_modules/gnuradio/scapy_radio/
  DEPENDS scapy_radio_python)

install(TARGETS scapy_radio_python DESTINATION ${GR_PYTHON_DIR}/gnuradio/scapy_radio COMPONENT pythonapi)
